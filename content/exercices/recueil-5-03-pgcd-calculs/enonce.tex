\begin{itemize}
  \item Soit $n$ un entier. Calculer $\mathrm{PGCD}(5n+2,\ 12n+5)$ et
  $\mathrm{PGCD}(9n+4,\ 2n-1)$.
  \item Soient $a$ et $b$ deux entiers non tous nuls. Supposons qu'ils
  existent trois entiers $c, x$ et $y$ tels que $c = ax + by$. Alors, on a :
  $\mathrm{PGCD}(a, b)$ divise $c$.
\end{itemize}
